**2. Create `__tests__/integration/integration.test.ts`** with 3 describe blocks:

```typescript
import * as fs from 'fs';
import * as path from 'path';
import { waitForN8n, getExpectedNodes, N8N_BASE_URL, N8N_INTERNAL_TYPES_URL } from './helpers';

const FIXTURES_DIR = path.join(__dirname, 'fixtures');

describe('n8n Integration Tests', () => {

  beforeAll(async () => {
    // Wait for n8n to be healthy (container started by integration-test.sh)
    await waitForN8n(N8N_BASE_URL, 60000);
  }, 90000); // 90s timeout for beforeAll

  describe('Node Registration (INT-01)', () => {
    let registeredNodeNames: string[];

    beforeAll(async () => {
      const response = await fetch(N8N_INTERNAL_TYPES_URL);
      expect(response.ok).toBe(true);
      const nodeTypes = await response.json();
      registeredNodeNames = nodeTypes.map((n: any) => n.name);
    });

    const expectedNodes = getExpectedNodes();

    // Generate a test per package
    for (const pkg of expectedNodes) {
      for (const nodeType of pkg.nodeTypes) {
        it(`should have registered ${nodeType}`, () => {
          expect(registeredNodeNames).toContain(nodeType);
        });
      }
    }
  });

  describe('Credential Registration (INT-02)', () => {
    let registeredCredentialNames: string[];

    beforeAll(async () => {
      // /types/credentials.json is the internal endpoint for credentials
      const response = await fetch(`${N8N_BASE_URL}/types/credentials.json`);
      expect(response.ok).toBe(true);
      const credentialTypes = await response.json();
      registeredCredentialNames = credentialTypes.map((c: any) => c.name);
    });

    const expectedNodes = getExpectedNodes();
    const expectedCredentials = expectedNodes
      .flatMap(n => n.credentialTypes)
      .filter(c => c.length > 0);

    for (const credType of expectedCredentials) {
      it(`should have registered credential ${credType}`, () => {
        expect(registeredCredentialNames).toContain(credType);
      });
    }
  });

  describe('Smoke Workflow Import (INT-03)', () => {
    const fixtureFiles = fs.readdirSync(FIXTURES_DIR)
      .filter(f => f.startsWith('workflow-') && f.endsWith('.json'));

    for (const fixtureFile of fixtureFiles) {
      it(`should import ${fixtureFile} successfully`, async () => {
        const workflowJson = JSON.parse(
          fs.readFileSync(path.join(FIXTURES_DIR, fixtureFile), 'utf-8')
        );
        const response = await fetch(`${N8N_BASE_URL}/api/v1/workflows`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(workflowJson),
        });
        // If auth is required, expect 200 or handle 401 gracefully
        // With N8N_USER_MANAGEMENT_DISABLED=true, API should work without auth
        expect(response.status).toBe(200);
        const result = await response.json();
        expect(result).toHaveProperty('id');
      });
    }
  });

});
```

IMPORTANT: The credential endpoint might be `/types/credentials.json` (similar pattern to `/types/nodes.json`). If this doesn't work during testing, fall back to checking credentials are listed in the node type definitions from `/types/nodes.json` (each node type has a `credentials` property listing the credential types it accepts).

IMPORTANT: The workflow import API may require authentication even with user management disabled. If `POST /api/v1/workflows` returns 401, try these approaches in order:
1. Check if the API works with `N8N_USER_MANAGEMENT_DISABLED=true` without any auth header
2. If 401, try programmatically creating an owner account via `POST /api/v1/owner/setup` with body `{email, firstName, lastName, password}` in beforeAll, then use the returned API key
3. If that also fails, skip the workflow import tests with a clear message

The test file should handle these cases gracefully in beforeAll setup.
