Reference pattern for correct nock test (from CEIDG):
```typescript
it('endpoint returns 200', () => {
  const scope = createNockScope(BASE_URL)
    .get('/path')
    .query({ param: 'value' })
    .reply(200, { data: [] });

  const https = require('https');
  return new Promise<void>((resolve) => {
    https.get(`${BASE_URL}/path?param=value`,
      (res: { statusCode: number; on: (event: string, cb: (chunk: string) => void) => void }) => {
        expect(res.statusCode).toBe(200);
        let data = '';
        res.on('data', (chunk: string) => { data += chunk; });
        res.on('end', () => {
          const body = JSON.parse(data);
          expect(body).toEqual({ data: [] });
          expect(scope.isDone()).toBe(true);
          resolve();
        });
      });
  });
});
```
